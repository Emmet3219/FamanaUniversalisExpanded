country_event = {
	id = estal.1
	title = estal.1.t
	desc = estal.1.d
	picture = VILLAGE_BURNING_eventPicture
	
	trigger = {
        OR = {
			exists = HSH
			exists = HOF
		}
		NOT = {
			has_country_modifier = yom_raided_timer
		}
        NOT = {
			technology_group = antestal_yom_tech_gr
		}
		any_owned_province = {
			OR = {
				region = west_estal_region
				region = napora_region
				region = west_ordica_region
				area = area_149
				area = area_148
				area = area_136
				area = area_105 
				area = feyel_area
				area = ratahja_area
			}
        }
	}
	
	mean_time_to_happen = {
		months = 30
	}

	immediate = {
        hidden_effect = {
			add_country_modifier = {
                name = yom_raided_timer
                duration = 1825
                hidden = yes
            }
            random_owned_province = { 
				limit = {
					OR = {
					    region = west_estal_region
					    region = napora_region
					    region = west_ordica_region
					    area = area_149
					    area = area_148
					    area = area_136
					    area = area_105 
					    area = feyel_area
					    area = ratahja_area
					}
				}
				add_devastation = 15
				save_event_target_as = prov_raided_yoms
			}
		}
	}
	
	option = {
		name = "estal.1.a"
		ai_chance = {
			factor = 3
			modifier = {
				factor = 0
				num_of_loans = 1
			}
			modifier = {
				factor = 0
				NOT = { years_of_income	= 0.25 }
			}
		}
		add_years_of_income = -0.25
		goto = prov_raided_yoms
	}
	option = {
		name = estal.1.b
		ai_chance = { factor = 1 }
		event_target:prov_raided_yoms = {
		    cancel_construction = yes
            add_local_autonomy = 15
		    add_devastation = 10
		}
		goto = prov_raided_yoms
	}
}

#Qurak (Erlaqi migration)
country_event = {
	id = estal.4
	title = estal.4.t
	desc = estal.4.t
	picture = LAND_MILITARY_eventPicture
	fire_only_once = yes
	hidden = yes
	
	trigger = {
		any_owned_province = {
            region = emastalia_region
        }
		NOT = { 
			exists = ELQ
		}
		6762 = {
			is_empty = yes
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		6762 = {
			cede_province = ELQ
            add_core = ELQ
			if = {
				limit = {
					ELQ = {
						primary_culture = Maxan_East_Emastalian
					}
				}
				change_culture = Maxan_East_Emastalian
			}
			if = {
				limit = {
					ELQ = {
						primary_culture = East_Emastalian
					}
				}
				change_culture = East_Emastalian
			}
            change_religion = lathvy 
		}
		emastalia_region = {
			remove_core = ELQ
			remove_claim = ELQ
		}
		west_emastalia_region = {
			remove_core = ELQ
			remove_claim = ELQ
		}
		ELQ = { 
			override_country_name = Qurak
			change_religion = lathvy
		}
		set_global_flag = qurak_happened
		
	}
	option = {
		name = estal.4.a
		ai_chance = { factor = 1 }
	}
}
